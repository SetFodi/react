import React, {useState} from 'react';
import './App.css'

function App(){
  const [count, setCount] = useState(0)

  const increase = () =>{
    setCount(count + 1)
  }

  const decrease = () =>{
    setCount(count - 1)
}

return(
  <div className='App'>
    <h1>Simple Counter Website</h1>
    <p>Counter: {count}</p>
    <button onClick={increase}>Increase</button>
    <button onClick={decrease}>Decrease</button>
  </div>
 )
}

export default App
